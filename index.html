import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import javax.swing.*;

public class Event extends JFrame implements ActionListener {
    JLabel l1, l2;
    JTextField tf1;
    JButton b1;

    Event() {
        l1 = new JLabel("Name");
        tf1 = new JTextField();
        b1 = new JButton("Display");
        l2 = new JLabel();
        b1.addActionListener(this);

        add(l1);
        add(tf1);
        add(b1);
        add(l2);

        setSize(300, 300);
        setLayout(new GridLayout(2, 2));
        setVisible(true);
    }

    public static void main(String[] args) {
        Event obj = new Event();
    }

    @Override
    public void actionPerformed(ActionEvent e) {
        String n = tf1.getText();
        l2.setText("Welcome " + n);
    }
}

import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import javax.swing.*;

public class Event extends JFrame implements ActionListener {
    JLabel l1, l2, l3;
    JTextField t1, t2, t3;
    JButton b;

    Event() {
        l1 = new JLabel("num1");
        l2 = new JLabel("num2");
        l3 = new JLabel("result");

        b = new JButton("result");

        t1 = new JTextField();
        t2 = new JTextField();
        t3 = new JTextField();

        b.addActionListener(this);

        add(l1);
        add(t1);

        add(l2);
        add(t2);

        add(b);

        add(l3);

        setSize(300, 300);
        setLayout(new GridLayout(3, 3));
        setVisible(true);
    }

    public static void main(String[] args) {
        Event obj = new Event();
    }

    @Override
    public void actionPerformed(ActionEvent e) {
        int a = Integer.parseInt(t1.getText());
        int b = Integer.parseInt(t2.getText());
        int c = a+b;
        t3.setText(""+c);
    }
}

import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import javax.swing.*;

public class Event extends JFrame {
    private JButton b;
    private JTextArea ta;

    public Event() {
        b = new JButton("Click");
        ta = new JTextArea(20, 20);

        b.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                ta.setText("Welcome");
            }
        });

        add(b);
        add(ta);
        
        setSize(300,300);
        setLayout(new GridLayout(2,1));
        
        setDefaultCloseOperation(EXIT_ON_CLOSE);
    }

    public static void main(String[] args) {
        Event obj = new Event();
        
        obj.setVisible(true);
    }
}
